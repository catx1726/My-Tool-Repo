起因:用 Promise 实现函数组合??!

否认，函数组合是同步，而 Promise 是异步

        
          async function fn1() {
            return await new Promise((resolve, reject)=> {
              resolve('fn1 ok!')
            })
          }

          async function fn2() {
            return
            await 'fn2 ok!'
          }
          
          Promise.race([fn1(), fn2()]).then((val) => {
            console.log('race:', val)
          })

          Promise.all([fn1(), fn2()]).then((val) => {
            console.log('all:', val)
          })
